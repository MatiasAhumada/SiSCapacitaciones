#!/bin/bash

echo "🚀 Deploying SiSCapacitaciones..."

cd /root/SiSCapacitaciones

# Pull cambios
git pull

# Backend
echo "📦 Building backend..."
cd back
npm install --production
npm run build
pm2 restart siscapacitaciones-api

# Frontend
echo "🎨 Building frontend..."
cd ../front
npm install
npm run build

# Copiar build a nginx
sudo rm -rf /var/www/siscapacitaciones/*
sudo cp -r dist/* /var/www/siscapacitaciones/
sudo chown -R www-data:www-data /var/www/siscapacitaciones
echo "✅ Deployment complete!"

server {
    server_name siscapacitaciones.net www.siscapacitaciones.net;

    location / {
        root /root/SiSCapacitaciones/front/dist;
        try_files $uri $uri/ /index.html;

        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header X-XSS-Protection "1; mode=block" always;
        
    }

    location /backend/ {
        proxy_pass http://127.0.0.1:4040/; # elimina /backend del path
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
        # Caché para archivos estáticos
    location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
        root /root/SiSCapacitaciones/front/dist;
        expires 1y;
        add_header Cache-Control "public, immutable";
    }

    listen 443 ssl;
    ssl_certificate /etc/letsencrypt/live/siscapacitaciones.net/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/siscapacitaciones.net/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;
}

server {
    if ($host = www.siscapacitaciones.net) {
        return 301 https://$host$request_uri;
    }

    if ($host = siscapacitaciones.net) {
        return 301 https://$host$request_uri;
    }

    listen 80;
    server_name siscapacitaciones.net www.siscapacitaciones.net;
    return 404;
}
